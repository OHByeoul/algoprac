   AO.NAME
FROM 
    ANIMAL_OUTS AO LEFT OUTER JOIN ANIMAL_INS AI ON AI.ANIMAL_ID = AO.ANIMAL_ID

ORDER BY AO.ANIMAL_ID;

오랜기간 보호한 동물

-- 코드를 입력하세요
SELECT 
     TB.NAME,
    TB.INDATE
FROM (SELECT 
    ROW_NUMBER() OVER(ORDER BY NVL(AO.DATETIME,SYSDATE)-AI.DATETIME DESC) AS RN, 
    AI.NAME,
    AI.DATETIME AS INDATE,
    AO.DATETIME AS OUTDATE
    FROM ANIMAL_INS AI LEFT OUTER JOIN ANIMAL_OUTS AO ON AI.ANIMAL_ID = AO.ANIMAL_ID) TB
WHERE RN <=3;
