상위 N개 레코드

SELECT
    NAME
FROM
    (SELECT
     ROWNUM AS RN,
        NAME
    FROM
    (SELECT
            NAME
        FROM ANIMAL_INS 
         ORDER BY DATETIME)
    ORDER BY RN)
WHERE RN = 1
