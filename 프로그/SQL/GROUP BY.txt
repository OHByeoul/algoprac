SELECT
    NAME,
    COUNT(NAME)
FROM
    ANIMAL_INS 
GROUP BY NAME
HAVING COUNT(NAME) > 1
ORDER BY NAME;

3. 입양시각

SELECT H.hour, count(1) count
FROM
    (SELECT
        TO_CHAR(datetime, 'HH24') hour
    FROM
        ANIMAL_OUTS) h
WHERE H.hour >= 9 AND H.hour <= 19
GROUP BY H.hour
ORDER BY H.hour

4. 입양시각2

SELECT 
    TB.HOUR,
    NVL(AO.COUNT,0)
FROM (
        SELECT (LEVEL-1) AS HOUR
        FROM DUAL
     CONNECT BY LEVEL <= 24
    ) TB
    LEFT OUTER JOIN
    (SELECT
        TO_CHAR(DATETIME,'HH24') AS HOUR,
        COUNT(TO_CHAR(DATETIME,'HH24')) AS COUNT
    FROM 
        ANIMAL_OUTS 
    GROUP BY 
         TO_CHAR(DATETIME,'HH24')) AO
ON TB.HOUR = AO.HOUR
ORDER BY TB.HOUR
